```javascript
function (event) {
  var viewModel = this;
  //设置保存前校验
  viewModel.on("beforeSave", function(args){
      var code = viewModel.get("code").getValue();
      if(code === undefined || code === null){
        cb.utils.alert("编码不能为空！");
        return false;
      }
  });
}
```
